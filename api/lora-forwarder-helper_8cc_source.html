82 Ptr<Application>
83 LoraForwarderHelper::InstallPriv(Ptr<Node> node) const
84 {
85  NS_LOG_FUNCTION(this << node);
86 
87  Ptr<LoraForwarder> app = m_factory.Create<LoraForwarder>();
88 
89  app->SetNode(node);
90  node->AddApplication(app);
91 
92  // Link the LoraForwarder to the SatLorawanNetDevices
93  for (uint32_t i = 0; i < node->GetNDevices(); i++)
94  {
95  Ptr<NetDevice> currentNetDevice = node->GetDevice(i);
96  if (currentNetDevice->GetObject<SatLorawanNetDevice>() != nullptr)
97  {
98  Ptr<SatLorawanNetDevice> loraNetDevice =
99  currentNetDevice->GetObject<SatLorawanNetDevice>();
100  uint8_t beamId = loraNetDevice->GetLorawanMac()->GetBeamId();
101  app->SetLoraNetDevice(beamId, loraNetDevice);
102  loraNetDevice->SetReceiveNetworkServerCallback(
103  MakeCallback(&LoraForwarder::ReceiveFromLora, app));
104  }
105  else if (currentNetDevice->GetObject<PointToPointNetDevice>() != nullptr)
106  {
107  Ptr<PointToPointNetDevice> pointToPointNetDevice =
108  currentNetDevice->GetObject<PointToPointNetDevice>();
109  app->SetPointToPointNetDevice(pointToPointNetDevice);
110  pointToPointNetDevice->SetReceiveCallback(
111  MakeCallback(&LoraForwarder::ReceiveFromPointToPoint, app));
112  }
113  }
114 
115  return app;
116 }

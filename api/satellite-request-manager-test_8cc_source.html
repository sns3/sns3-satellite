163 SatBaseTestCase::SendControlMsg(Ptr<SatControlMessage> msg, const Address& dest)
164 {
165  if (msg->GetMsgType() == SatControlMsgTag::SAT_CR_CTRL_MSG)

172  m_rcvdCapacityRequests.push_back(cr);

177  }
